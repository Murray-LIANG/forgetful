# get token
curl -i \
  -H "Content-Type: application/json" \
  -d '
    { "auth": {
        "identity": {
          "methods": ["password"],
          "password": {
            "user": {
              "name": "admin",
              "domain": { "id": "default" },
              "password": "welcome"
            }
          }
        },
        "scope": {
          "project": {
            "name": "admin",
            "domain": { "id": "default" }
          }
        }
      }
    }' \
  "http://192.168.77.63/identity/v3/auth/tokens"
